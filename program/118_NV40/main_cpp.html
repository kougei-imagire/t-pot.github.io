0174: VOID WINAPI ParaboloidJacobian (D3DXVECTOR4* pOut, CONST D3DXVECTOR2* pTexCoord, 
0175: CONST D3DXVECTOR2* pTexelSize, LPVOID pData)
0176: {
0177:     UNREFERENCED_PARAMETER( pTexelSize );
0178: 
0179:     FLOAT x = 2.0f*( pTexCoord->x-0.5f);
0180:     FLOAT y = 2.0f*(-pTexCoord->y+0.5f);
0181:     
0182:     FLOAT r2 = x*x+y*y;
0183:     FLOAT J = 0.5f*(1.0f+r2);
0184: 
0185:     FLOAT col = (r2<(1.0f-1.0f/(FLOAT)MAP_SIZE)*(1.0f-1.0f/(FLOAT)MAP_SIZE)) ? J : 0.0f;
0186:     
0187:     // Normal vector
0188:     FLOAT nz = 0.5f * (1.0f-r2);
0189:     FLOAT nx = x;
0190:     FLOAT ny = y;
0191:     FLOAT n = sqrtf(nx*nx+ny*ny+nz*nz);
0192:     nx /= n;
0193:     ny /= n;
0194:     nz /= n;
0195: 
0196:     pOut->x = 0.5f * col * (-nx) + 0.5f;
0197:     pOut->y = 0.5f * col * ( ny) + 0.5f;
0198:     pOut->z = 0.5f * col * (-nz) + 0.5f;
0199:     pOut->w = 0.5f * col         + 0.5f;
0200: }

0205: VOID WINAPI FillTex (D3DXVECTOR4* pOut, CONST D3DXVECTOR2* pTexCoord, 
0206: CONST D3DXVECTOR2* pTexelSize, LPVOID pData)
0207: {
0208:     UNREFERENCED_PARAMETER( pData );
0209: 
0210:     FLOAT x = 2.0f*(pTexCoord->x-0.5f);
0211:     FLOAT y = 2.0f*(pTexCoord->y-0.5f);
0212:     FLOAT col = (x*x+y*y<(1.0f-pTexelSize->x)*(1.0f-pTexelSize->y))
0213:                 ? 1.0f : 0.0f;
0214:     
0215:     pOut->x = pOut->y = pOut->z = pOut->w = col;
0216: }
