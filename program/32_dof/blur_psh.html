0010: ; テクスチャーの色を引っ張ってくる
0011: tex t0      ; 0:0 0 0  1:0 0 0  2:0 0 0  3:0 0 0

0013: tex t2      ;   0 0 0    0 1 0    0 0 1    0 0 0  
0014: tex t3

0016: ; r0 = 0.5*(0.5*(t0+t1)+0.5*(t2+t3))) = (t0+t1+t2+t3)/4
0017: ; 色の平均を取ってくる

0019: lrp r1, c0, t2, t3
